CREATE TABLE ROL (
idRol INT IDENTITY PRIMARY KEY NOT NULL,
nombre VARCHAR(20) NOT NULL
)

INSERT INTO ROL (nombre) VALUES ('ADMINISTRADOR')
INSERT INTO ROL (nombre) VALUES ('USUARIO')

CREATE TABLE USUARIO (
idUsuario INT IDENTITY PRIMARY KEY NOT NULL, 
nombre VARCHAR(20) NOT NULL,
apellido VARCHAR(20) NOT NULL,
usuario VARCHAR(10) NOT NULL,
passwordHash nVARCHAR(64) NOT NULL,
rol INT FOREIGN KEY REFERENCES ROL(idRol) NOT NULL,
activo BIT NOT NULL
)

CREATE TABLE CAJAS (
idCaja INT IDENTITY PRIMARY KEY NOT NULL,
idCajero INT FOREIGN KEY REFERENCES USUARIO(idUsuario) NOT NULL,
disponibilidad BIT NOT NULL
)
